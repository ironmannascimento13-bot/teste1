import yfinance as yf import pandas as pd import numpy as np import statsmodels.api as sm import matplotlib.pyplot as plt # ---------------------------------- # 1) BAIXA DADOS # ---------------------------------- start = "2024-01-01" end = "2026-02-28" # PETR4 petr = yf.download("PETR4.SA", start=start, end=end) # Brent (futuro) – ticker do Yahoo Finance brent = yf.download("BZ=F", start=start, end=end) # IBOV ibov = yf.download("^BVSP", start=start, end=end) # USD/BRL do Yahoo fx = yf.download("USDBRL=X", start=start, end=end) # Ajuste de preços df = pd.DataFrame({ "PETR4": petr["Adj Close"], "Brent": brent["Close"], "IBOV": ibov["Adj Close"], "USD/BRL": fx["Close"] }) df.dropna(inplace=True) # ---------------------------------- # 2) RETORNOS # ---------------------------------- df["r_PETR4"] = np.log(df["PETR4"] / df["PETR4"].shift(1)) df["r_Brent"] = np.log(df["Brent"] / df["Brent"].shift(1)) df["r_IBOV"] = np.log(df["IBOV"] / df["IBOV"].shift(1)) df["r_FX"] = np.log(df["USD/BRL"] / df["USD/BRL"].shift(1)) df.dropna(inplace=True) # ---------------------------------- # 3) BETAS EWLS (decay 0.96) # ---------------------------------- decay = 0.96 # Weights (exponencial) def ewls_weights(n): w = decay ** np.arange(n)[::-1] return w / w.sum() # Função EWLS beta def ewls_beta(y, x): w = ewls_weights(len(y)) X = sm.add_constant(x) model = sm.WLS(y, X, weights=w).fit() return model.params # Inicializa betas df["beta_oil"] = np.nan df["beta_ibov"] = np.nan df["beta_fx"] = np.nan window = 60 for i in range(window, len(df)): y = df["r_PETR4"].iloc[i-window:i] X1 = df["r_Brent"].iloc[i-window:i] X2 = df["r_IBOV"].iloc[i-window:i] X3 = df["r_FX"].iloc[i-window:i] params = ewls_beta(y, pd.concat([X1, X2, X3], axis=1)) df.loc[df.index[i], ["beta_oil","beta_ibov","beta_fx"]] = params[1:] # Residual narrativo df["u_t"] = df["r_PETR4"] - (df["beta_oil"]*df["r_Brent"] + df["beta_ibov"]*df["r_IBOV"] + df["beta_fx"]*df["r_FX"]) df.dropna(inplace=True) # ---------------------------------- # 4) EVENTOS (lista exemplo) # ---------------------------------- # Substitua pelas datas reais e buckets que você já organizou events = pd.DataFrame([ {"date":"2025-05-12","type":"dividendo"}, {"date":"2025-08-07","type":"dividendo"}, {"date":"2025-11-06","type":"dividendo"}, {"date":"2024-11-21","type":"plano"}, {"date":"2025-11-28","type":"plano"}, {"date":"2024-05-10","type":"politico"}, ], dtype=str) events["date"] = pd.to_datetime(events["date"]) # Calcula Shock/Drift records = [] for _, row in events.iterrows(): t0 = row["date"] if t0 in df.index: shock = df["u_t"].loc[t0 - pd.Timedelta("1D"):t0 + pd.Timedelta("1D")].mean() drift = df["u_t"].loc[t0 + pd.Timedelta("1D"):t0 + pd.Timedelta("10D")].sum() records.append({"date":t0, "type":row["type"], "shock":shock, "drift":drift}) evt_df = pd.DataFrame(records) # ---------------------------------- # 5) REGRESSÃO θ # ---------------------------------- print("\nθ por tipo de evento:") for t in evt_df["type"].unique(): sub = evt_df[evt_df["type"]==t] if len(sub)>=3: X = sm.add_constant(sub["shock"]) y = sub["drift"] r = sm.OLS(y, X).fit() print(f"Tipo {t} — θ (coef shock): {r.params['shock']:.4f}") # ---------------------------------- # 6) GRÁFICOS # ---------------------------------- plt.figure(figsize=(12,6)) # (a) Preço PETR4 plt.subplot(2,1,1) plt.plot(df["PETR4"], label="PETR4") plt.title("PETR4 – Preço Ajustado") plt.legend() # (b) Residual narrativo u_t plt.subplot(2,1,2) plt.plot(df["u_t"], label="Residual narrativo") plt.axhline(0, color="black", linestyle="--") plt.title("Residual narrativo (u_t)") plt.legend() plt.tight_layout() plt.show() print("\nEventos Shock/Drift:") print(evt_df)
